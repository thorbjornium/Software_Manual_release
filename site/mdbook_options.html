Include other files

Include line 2-10 of the file rust.rs.

```{{#include rust.rs:2:10}}```

fn do_things<T>(a: T, b: T) where
    T: IntoIterator,
    T::Item: Display {
    for (e,f) in a.into_iter().zip(b.into_iter()) {
        println!("{} {}", e, f);
    }
}
